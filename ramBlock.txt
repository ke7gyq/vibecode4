#define FLASH_SIZE 4096
#define PERSISTANT_AREA FLASH_END - FLASH_SIZE


// Setter for cerdentials
setWan ( char * ssid, password ) 
	struct (PersistantMemory *) ramMem = malloc ( FLASH_SIZE) ;
	memcpy ( ramMem->wan.ssid, ssid);
	memcpy ( ramMem->wan.password, password);
	flash_write ( PERSISTANT_AREA, ramMem, FLASH_SIZE );
	free ( ramMem)
}

	struct (PersistantMemory *) p = PERSISTANT_AREA FLASH_END - FLASH_SIZE;
